create table trips ( id uuid primary key default uuid_generate_v4(), code text unique not null, name text not null, start_date date not null, end_date date not null, orcamento_previsto numeric(10,2), admin_id uuid references auth.users(id), deleted_at timestamptz default null, created_at timestamptz default now() ); create table families ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, name text not null, pin_hash text not null, created_by uuid references auth.users(id), created_at timestamptz default now() ); create table members ( id uuid primary key default uuid_generate_v4(), family_id uuid references families(id) on delete cascade, trip_id uuid references trips(id) on delete cascade, name text not null, pix_key text, cpf_file_url text, rg_file_url text, emergencia_nome text, emergencia_fone text, info_medica text, created_by uuid references auth.users(id), created_at timestamptz default now() ); create table transfers ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, data_hora timestamptz not null, origem text not null, destino text not null, responsavel text, contato text, placa text, obs text, created_by uuid references auth.users(id), created_at timestamptz default now() ); create table transport_providers ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, empresa text not null, whatsapp text not null, tipo_veiculo text, capacidade int, responsavel text, created_by uuid references auth.users(id), created_at timestamptz default now() ); create table transfer_passengers ( transfer_id uuid references transfers(id) on delete cascade, member_id uuid references members(id) on delete cascade, primary key (transfer_id, member_id) ); create table hotels ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, nome text not null, endereco text, contato text, checkin date, checkout date, voucher_url text, created_by uuid references auth.users(id), created_at timestamptz default now() ); create table hotel_guests ( hotel_id uuid references hotels(id) on delete cascade, member_id uuid references members(id) on delete cascade, quarto text, primary key (hotel_id, member_id) ); create table flights ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, origem text, destino text, data date, cia text, numero_voo text, horario text, file_url text, created_at timestamptz default now() ); create table flight_passengers ( flight_id uuid references flights(id) on delete cascade, member_id uuid references members(id) on delete cascade, primary key (flight_id, member_id) ); create table flight_tickets ( id uuid primary key default uuid_generate_v4(), flight_id uuid references flights(id) on delete cascade, member_id uuid references members(id) on delete cascade, ticket_url text not null, localizador text, assento text, uploaded_by uuid references auth.users(id), created_at timestamptz default now(), unique(flight_id, member_id) ); create table parks ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, nome text not null, data date, created_at timestamptz default now() ); create table park_tickets ( park_id uuid references parks(id) on delete cascade, member_id uuid references members(id) on delete cascade, ticket_url text not null, primary key (park_id, member_id) ); create table attractions ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, nome text not null, data_hora timestamptz, created_at timestamptz default now() ); create table attraction_tickets ( attraction_id uuid references attractions(id) on delete cascade, member_id uuid references members(id) on delete cascade, ticket_url text not null, primary key (attraction_id, member_id) ); create table alarms ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, titulo text not null, descricao text, data_hora timestamptz not null, created_by uuid references auth.users(id), created_at timestamptz default now() ); create table expenses ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, descricao text not null, valor numeric(10,2) not null, data date not null, categoria text, comprovante_url text, pago_por uuid references members(id), created_by uuid references auth.users(id), created_at timestamptz default now() ); create table expense_splits ( expense_id uuid references expenses(id) on delete cascade, member_id uuid references members(id) on delete cascade, valor_dividido numeric(10,2) not null, quitado boolean default false, primary key (expense_id, member_id) ); create table messages ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, user_id uuid references auth.users(id), member_name text, texto text not null, created_at timestamptz default now() ); create table polls ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, pergunta text not null, created_by uuid references auth.users(id), created_at timestamptz default now() ); create table poll_options ( id uuid primary key default uuid_generate_v4(), poll_id uuid references polls(id) on delete cascade, texto text not null ); create table poll_votes ( poll_id uuid references polls(id) on delete cascade, option_id uuid references poll_options(id) on delete cascade, user_id uuid references auth.users(id), primary key (poll_id, user_id) ); create table gallery ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, url text not null, descricao text, uploaded_by_member_id uuid references members(id), created_at timestamptz default now() ); create table documents ( id uuid primary key default uuid_generate_v4(), member_id uuid references members(id) on delete cascade, family_id uuid references families(id) on delete cascade, tipo text, arquivo_url text not null, voo_id uuid references flights(id), atracao_nome text, created_at timestamptz default now() ); create table roteiro ( trip_id uuid references trips(id) on delete cascade primary key, content_md text, updated_at timestamptz default now() ); create table checkins_localizacao ( id uuid primary key default uuid_generate_v4(), trip_id uuid references trips(id) on delete cascade, member_id uuid references members(id) on delete cascade, lat numeric, lng numeric, pin_temporario text, expira_em timestamptz, created_at timestamptz default now(), unique(trip_id, member_id) ); create or replace view public_flight_summary as select f.id, f.trip_id, f.origem, f.destino, f.data, f.horario, fam.name as family_name, count(ft.member_id) as quantidade from flights f left join flight_tickets ft on ft.flight_id = f.id left join members m on m.id = ft.member_id left join families fam on fam.id = m.family_id group by f.id, f.trip_id, f.origem, f.destino, f.data, f.horario, fam.name;
